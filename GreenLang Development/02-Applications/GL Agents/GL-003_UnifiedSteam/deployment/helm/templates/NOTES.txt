--------------------------------------------------------------------------------
  QUICK START
--------------------------------------------------------------------------------

1. Check deployment status:
   kubectl get pods -n {{ .Release.Namespace }} -l app.kubernetes.io/name={{ include "unifiedsteam.name" . }}

2. View logs:
   kubectl logs -n {{ .Release.Namespace }} -l app.kubernetes.io/name={{ include "unifiedsteam.name" . }} -f

3. Access API documentation:
   kubectl port-forward -n {{ .Release.Namespace }} svc/{{ include "unifiedsteam.fullname" . }} 8080:{{ .Values.service.http.port }}
   Then open: http://localhost:8080/docs

4. Access Prometheus metrics:
   kubectl port-forward -n {{ .Release.Namespace }} svc/{{ include "unifiedsteam.fullname" . }} 9090:{{ .Values.service.metrics.port }}
   Then open: http://localhost:9090/metrics

5. Test health endpoint:
   kubectl exec -n {{ .Release.Namespace }} -it deploy/{{ include "unifiedsteam.fullname" . }} -- curl http://localhost:{{ .Values.containerPorts.http }}/health
